72 uint16_t uart_ubr(uint32_t baud, int *u2x, uint32_t *actual)
73 {
74  double div;
75  uint32_t ubr_regi;
76 
77  // Calculating Baud Rate
78  // (U2X = 0)
79  // BAUD = Fosc/(16*(UBRn+1))
80  // UBRn = Fosc/(16*Baud) -1
81  // (U2X = 1)
82  // BAUD = Fosc/(8*(UBRn+1))
83  // UBRn = Fosc/(8*Baud) -1
84 
86  *u2x = 1;
87  div = 8;
88  ubr_regi = round( ((double)F_CPU/(div*(double)baud)) - 1.0 );
89 
90  // For lower baud rates use 16 divider if the UBRR register overflows
91  // URBRR register is only a 12 bit register!
92  if(ubr_regi > 4095)
93  {
95  *u2x = 0;
96  div = 16.0;
97  ubr_regi = round( ((double)F_CPU/(div*(double)baud)) - 1.0 );
98  }
99  //overflow, baud rate was too low - so we clip to maximum allowed
100  if(ubr_regi > 4095)
101  ubr_regi = 4095;
102 
103  *actual = ((double)F_CPU/(div*((double)(ubr_regi+1))));
104 
105  return(ubr_regi);
106 }
